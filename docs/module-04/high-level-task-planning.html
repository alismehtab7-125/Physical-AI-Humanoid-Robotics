High-Level Task Planning using LLMs

Introduction to High-Level Task Planning​

High-level task planning involves converting abstract human commands into executable robot actions. This process requires understanding natural language, reasoning about the environment, and decomposing complex tasks into primitive actions that the robot can execute. Large Language Models (LLMs) have revolutionized this field by providing sophisticated natural language understanding and reasoning capabilities.

The Task Planning Pipeline​

The high-level task planning process typically follows this pipeline:

  1. Natural Language Understanding: Interpret the human command
  2. Environment Context Integration: Combine command with current world state
  3. Task Decomposition: Break down the task into subtasks
  4. Action Sequencing: Order actions logically
  5. Execution Monitoring: Track progress and handle failures

Planning Domain Definition Language (PDDL)​

Overview of PDDL​

PDDL (Planning Domain Definition Language) is a formal language for describing planning problems in AI. It provides a standardized way to represent:

  • Domain: Types, predicates, and actions available
  • Problem: Initial state and goal conditions
  • Plan: Sequence of actions to achieve the goal

PDDL Structure for Robotics​

class PDDLDomain:
    def __init__(self):
        self.domain_name = "humanoid_robot_domain"
        self.types = {
            "location": ["kitchen", "living_room", "bedroom", "office"],
            "object": ["cup", "bottle", "book", "phone"],
            "robot": ["humanoid"]
        }
        self.predicates = [
            "at(?robot - robot, ?loc - location)",
            "holding(?robot - robot, ?obj - object)",
            "object_at(?obj - object, ?loc - location)",
            "reachable(?loc - location)",
            "occupied(?loc - location)"
        ]
        self.actions = [
            self.navigate_action(),
            self.pick_action(),
            self.place_action()
        ]

    def navigate_action(self):
        return {
            "name": "navigate",
            "parameters": ["?robot - robot", "?from - location", "?to - location"],
            "preconditions": [
                "at(?robot, ?from)",
                "reachable(?to)",
                "not (= ?from ?to)"
            ],
            "effects": [
                "at(?robot, ?to)",
                "not at(?robot, ?from)"
            ]
        }

    def pick_action(self):
        return {
            "name": "pick",
            "parameters": ["?robot - robot", "?obj - object", "?loc - location"],
            "preconditions": [
                "at(?robot, ?loc)",
                "object_at(?obj, ?loc)",
                "not holding(?robot, ?obj)"
            ],
            "effects": [
                "holding(?robot, ?obj)",
                "not object_at(?obj, ?loc)"
            ]
        }

    def place_action(self):
        return {
            "name": "place",
            "parameters": ["?robot - robot", "?obj - object", "?loc - location"],
            "preconditions": [
                "at(?robot, ?loc)",
                "holding(?robot, ?obj)"
            ],
            "effects": [
                "object_at(?obj, ?loc)",
                "not holding(?robot, ?obj)"
            ]
        }

PDDL Planning with External Solvers​

import subprocess
import tempfile
import os

class PDDLPlanner:
    def __init__(self, planner_path="fast-downward"):
        self.planner_path = planner_path

    def solve_problem(self, domain, problem):
        """
        Solve PDDL problem using external planner
        """
        # Write domain and problem to temporary files
        with tempfile.NamedTemporaryFile(mode='w', suffix='.pddl', delete=False) as domain_file:
            domain_content = self.write_domain_file(domain)
            domain_file.write(domain_content)
            domain_path = domain_file.name

        with tempfile.NamedTemporaryFile(mode='w', suffix='.pddl', delete=False) as problem_file:
            problem_content = self.write_problem_file(problem)
            problem_file.write(problem_content)
            problem_path = problem_file.name

        try:
            # Call external planner
            result = subprocess.run([
                self.planner_path,
                domain_path, problem_path,
                "--search", "astar(lmcut())"
            ], capture_output=True, text=True)

            if result.returncode == 0:
                plan = self.parse_plan(result.stdout)
            else:
                plan = None

        finally:
            # Clean up temporary files
            os.unlink(domain_path)
            os.unlink(problem_path)

        return plan

    def write_domain_file(self, domain):
        """
        Write PDDL domain file content
        """
        content = f"(define (domain {domain.domain_name})\n"
        content += "  (:requirements :strips :typing)\n"

        # Types
        content += "  (:types\n"
        for type_name, instances in domain.types.items():
            content += f"    {' '.join(instances)} - {type_name}\n"
        content += "  )\n"

        # Predicates
        content += "  (:predicates\n"
        for pred in domain.predicates:
            content += f"    ({pred})\n"
        content += "  )\n"

        # Actions
        for action in domain.actions:
            content += f"  (:action {action['name']}\n"
            content += f"    :parameters ({' '.join(action['parameters'])})\n"
            content += "    :precondition (and\n"
            for precond in action['preconditions']:
                content += f"      ({precond})\n"
            content += "    )\n"
            content += "    :effect (and\n"
            for effect in action['effects']:
                content += f"      ({effect})\n"
            content += "    )\n"
            content += "  )\n"

        content += ")\n"
        return content

    def write_problem_file(self, problem):
        """
        Write PDDL problem file content
        """
        content = f"(define (problem task)\n"
        content += f"  (:domain {problem.domain.domain_name})\n"

        # Objects
        content += "  (:objects\n"
        for obj_name, obj_type in problem.objects.items():
            content += f"    {obj_name} - {obj_type}\n"
        content += "  )\n"

        # Initial state
        content += "  (:init\n"
        for state in problem.initial_state:
            content += f"    ({state})\n"
        content += "  )\n"

        # Goal
        content += "  (:goal (and\n"
        for goal in problem.goal:
            content += f"    ({goal})\n"
        content += "  ))\n"
        content += ")\n"
        return content

    def parse_plan(self, plan_output):
        """
        Parse planner output into action sequence
        """
        lines = plan_output.split('\n')
        plan = []
        for line in lines:
            if line.strip().startswith('(') and line.strip().endswith(')'):
                action = line.strip()[1:-1]  # Remove parentheses
                plan.append(action)
        return plan

LLM-Guided Task Planning​

Natural Language to PDDL Translation​

LLMs can bridge the gap between natural language and formal planning languages:

import openai
from typing import Dict, List, Tuple

class LLMTaskPlanner:
    def __init__(self, llm_model="gpt-3.5-turbo"):
        self.llm_model = llm_model
        self.pddl_planner = PDDLPlanner()

    def plan_from_natural_language(self, command: str, world_state: Dict) -> List[str]:
        """
        Plan task from natural language command using LLM guidance
        """
        # Step 1: Parse natural language to structured representation
        structured_task = self.parse_natural_language(command, world_state)

        # Step 2: Convert to PDDL domain and problem
        pddl_domain = self.create_pddl_domain(structured_task)
        pddl_problem = self.create_pddl_problem(structured_task, world_state)

        # Step 3: Solve using PDDL planner
        plan = self.pddl_planner.solve_problem(pddl_domain, pddl_problem)

        # Step 4: Refine plan using LLM for better executability
        refined_plan = self.refine_plan_with_llm(plan, command, world_state)

        return refined_plan

    def parse_natural_language(self, command: str, world_state: Dict) -> Dict:
        """
        Parse natural language command into structured representation
        """
        prompt = f"""
        Parse the following natural language command into structured task representation:

        Command: "{command}"
        World State: {world_state}

        Provide the structured representation with:
        1. Main task objective
        2. Required objects
        3. Target locations
        4. Prerequisites
        5. Expected outcome

        Format the response as JSON.
        """

        response = openai.ChatCompletion.create(
            model=self.llm_model,
            messages=[{"role": "user", "content": prompt}]
        )

        import json
        return json.loads(response.choices[0].message.content)

    def create_pddl_domain(self, structured_task: Dict) -> PDDLDomain:
        """
        Create PDDL domain from structured task
        """
        domain = PDDLDomain()

        # Customize domain based on task requirements
        if "navigation" in structured_task.get("task_type", ""):
            # Add navigation-specific predicates and actions
            domain.predicates.extend([
                "traversable(?from - location, ?to - location)",
                "path_exists(?from - location, ?to - location)"
            ])

        if "manipulation" in structured_task.get("task_type", ""):
            # Add manipulation-specific predicates and actions
            domain.predicates.extend([
                "graspable(?obj - object)",
                "reachable(?obj - object)",
                "free_hand(?robot - robot)"
            ])

        return domain

    def create_pddl_problem(self, structured_task: Dict, world_state: Dict) -> PDDLProblem:
        """
        Create PDDL problem from structured task and world state
        """
        domain = self.create_pddl_domain(structured_task)

        # Create problem object
        problem = PDDLProblem(domain)

        # Set objects based on world state
        problem.objects = self.extract_objects_from_world_state(world_state)

        # Set initial state
        problem.initial_state = self.extract_initial_state(world_state)

        # Set goal state
        problem.goal = self.extract_goal_state(structured_task)

        return problem

    def extract_objects_from_world_state(self, world_state: Dict) -> Dict:
        """
        Extract objects from world state
        """
        objects = {}
        # Extract from world state representation
        for obj_name, obj_info in world_state.get("objects", {}).items():
            obj_type = obj_info.get("type", "object")
            objects[obj_name] = obj_type

        for loc_name, loc_info in world_state.get("locations", {}).items():
            objects[loc_name] = "location"

        # Add robot
        objects["robot1"] = "robot"

        return objects

    def extract_initial_state(self, world_state: Dict) -> List[str]:
        """
        Extract initial state from world state
        """
        initial_state = []

        # Robot location
        robot_pos = world_state.get("robot", {}).get("location", "unknown")
        if robot_pos != "unknown":
            initial_state.append(f"at(robot1, {robot_pos})")

        # Object locations
        for obj_name, obj_info in world_state.get("objects", {}).items():
            obj_loc = obj_info.get("location", "unknown")
            if obj_loc != "unknown":
                initial_state.append(f"object_at({obj_name}, {obj_loc})")

        # Reachable locations
        for loc_name in world_state.get("locations", {}).keys():
            initial_state.append(f"reachable({loc_name})")

        return initial_state

    def extract_goal_state(self, structured_task: Dict) -> List[str]:
        """
        Extract goal state from structured task
        """
        goal_state = []

        # Parse task objective
        objective = structured_task.get("objective", "")
        if "bring" in objective.lower() or "fetch" in objective.lower():
            # Goal: robot has object at target location
            target_loc = structured_task.get("target_location", "unknown")
            obj = structured_task.get("required_object", "unknown")
            if target_loc != "unknown" and obj != "unknown":
                goal_state.append(f"object_at({obj}, {target_loc})")

        elif "go to" in objective.lower() or "navigate to" in objective.lower():
            # Goal: robot at target location
            target_loc = structured_task.get("target_location", "unknown")
            if target_loc != "unknown":
                goal_state.append(f"at(robot1, {target_loc})")

        return goal_state

    def refine_plan_with_llm(self, plan: List[str], command: str, world_state: Dict) -> List[str]:
        """
        Refine the PDDL-generated plan using LLM for better executability
        """
        if not plan:
            return []

        prompt = f"""
        The following plan was generated for the command "{command}":
        {plan}

        World state: {world_state}

        Refine this plan to make it more executable by a humanoid robot. Consider:
        1. Physical constraints of humanoid locomotion
        2. Grasping and manipulation requirements
        3. Safety considerations
        4. Natural human-like behavior

        Return the refined plan as a list of executable actions.
        """

        response = openai.ChatCompletion.create(
            model=self.llm_model,
            messages=[{"role": "user", "content": prompt}]
        )

        import json
        try:
            refined_plan = json.loads(response.choices[0].message.content)
            return refined_plan if isinstance(refined_plan, list) else plan
        except:
            # If parsing fails, return original plan
            return plan

Hierarchical Task Networks (HTN)​

HTN vs PDDL​

While PDDL focuses on state-based planning, Hierarchical Task Networks provide a more natural way to represent complex tasks by decomposing them hierarchically:

class HTNPlanner:
    def __init__(self):
        self.tasks = {
            "fetch_object": self.decompose_fetch_object,
            "navigate_to_room": self.decompose_navigate_to_room,
            "manipulate_object": self.decompose_manipulate_object
        }

    def decompose_fetch_object(self, params):
        """
        Decompose fetch object task
        """
        return [
            ("navigate_to_room", {"room": params["object_location"]}),
            ("locate_object", {"object": params["object_name"]}),
            ("grasp_object", {"object": params["object_name"]}),
            ("navigate_to_room", {"room": params["delivery_location"]}),
            ("place_object", {"object": params["object_name"], "location": params["delivery_location"]})
        ]

    def decompose_navigate_to_room(self, params):
        """
        Decompose navigate to room task
        """
        return [
            ("find_path", {"destination": params["room"]}),
            ("execute_path", {"destination": params["room"]}),
            ("verify_arrival", {"location": params["room"]})
        ]

    def plan_with_htn(self, task, params):
        """
        Plan using HTN decomposition
        """
        if task in self.tasks:
            subtasks = self.tasks[task](params)
            plan = []
            for subtask, subparams in subtasks:
                if subtask in self.tasks:
                    plan.extend(self.plan_with_htn(subtask, subparams))
                else:
                    plan.append((subtask, subparams))
            return plan
        else:
            return [(task, params)]

import rclpy
from rclpy.node import Node
from std_msgs.msg import String
from geometry_msgs.msg import PoseStamped

from ros2_task_planning_interfaces.srv import PlanTask
from ros2_task_planning_interfaces.action import ExecuteTask

class LLMTaskPlanningNode(Node):
    def __init__(self):
        super().__init__('llm_task_planning_node')

        # Initialize LLM planner
        self.llm_planner = LLMTaskPlanner()

        # Service for task planning
        self.plan_service = self.create_service(
            PlanTask, 'plan_task', self.plan_task_callback
        )

        # Action server for task execution
        self.task_server = ActionServer(
            self,
            ExecuteTask,
            'execute_task',
            self.execute_task_callback
        )

        # Publishers for visualization
        self.plan_pub = self.create_publisher(String, 'planned_tasks', 10)
        self.status_pub = self.create_publisher(String, 'task_status', 10)

        # World state subscriber
        self.world_state_sub = self.create_subscription(
            String, 'world_state', self.world_state_callback, 10
        )

        self.current_world_state = {}

    def plan_task_callback(self, request, response):
        """
        Service callback for task planning
        """
        try:
            # Get current world state
            world_state = self.current_world_state

            # Plan task using LLM
            plan = self.llm_planner.plan_from_natural_language(
                request.command, world_state
            )

            response.plan = plan
            response.success = len(plan) > 0
            response.error_message = "" if response.success else "Failed to generate plan"

            # Publish plan for visualization
            plan_msg = String()
            plan_msg.data = str(plan)
            self.plan_pub.publish(plan_msg)

        except Exception as e:
            response.success = False
            response.error_message = str(e)
            self.get_logger().error(f"Task planning failed: {e}")

        return response

    def execute_task_callback(self, goal_handle):
        """
        Action callback for task execution
        """
        command = goal_handle.request.command
        self.get_logger().info(f'Executing task: {command}')

        # Get current world state
        world_state = self.current_world_state

        # Plan the task
        plan = self.llm_planner.plan_from_natural_language(command, world_state)

        if not plan:
            goal_handle.abort()
            result = ExecuteTask.Result()
            result.success = False
            result.message = "Could not generate plan"
            return result

        # Execute the plan step by step
        for i, action in enumerate(plan):
            if goal_handle.is_cancel_requested:
                goal_handle.canceled()
                result = ExecuteTask.Result()
                result.success = False
                result.message = "Task canceled"
                return result

            # Execute individual action
            action_success = self.execute_action(action)

            if not action_success:
                goal_handle.abort()
                result = ExecuteTask.Result()
                result.success = False
                result.message = f"Action failed: {action}"
                return result

            # Update feedback
            feedback = ExecuteTask.Feedback()
            feedback.current_action = str(action)
            feedback.progress = float(i + 1) / len(plan)
            goal_handle.publish_feedback(feedback)

        # Task completed successfully
        goal_handle.succeed()
        result = ExecuteTask.Result()
        result.success = True
        result.message = "Task completed successfully"
        return result

    def execute_action(self, action):
        """
        Execute a single action primitive
        """
        # This would interface with the robot's action execution system
        # For simulation, we'll return success
        self.get_logger().info(f'Executing action: {action}')
        return True

    def world_state_callback(self, msg):
        """
        Update world state from message
        """
        import json
        try:
            self.current_world_state = json.loads(msg.data)
        except:
            self.get_logger().warn("Could not parse world state message")

Handling Uncertainty and Failures​

Plan Monitoring and Recovery​

class AdaptiveTaskPlanner:
    def __init__(self, base_planner):
        self.base_planner = base_planner
        self.failure_recovery = {
            "navigation_failure": self.handle_navigation_failure,
            "grasping_failure": self.handle_grasping_failure,
            "object_not_found": self.handle_object_not_found
        }

    def execute_with_monitoring(self, command, world_state):
        """
        Execute plan with monitoring and recovery
        """
        plan = self.base_planner.plan_from_natural_language(command, world_state)

        for i, action in enumerate(plan):
            try:
                result = self.execute_action_with_monitoring(action)

                if not result.success:
                    # Handle failure
                    recovery_plan = self.handle_failure(result.error_type, action, plan[i+1:])
                    plan = plan[:i] + recovery_plan + plan[i+1:]

            except Exception as e:
                # Unexpected failure
                recovery_plan = self.handle_unexpected_failure(str(e), action, plan[i+1:])
                plan = plan[:i] + recovery_plan + plan[i+1:]

        return plan

    def handle_failure(self, error_type, failed_action, remaining_plan):
        """
        Generate recovery plan for specific failure type
        """
        if error_type in self.failure_recovery:
            return self.failure_recovery[error_type](failed_action, remaining_plan)
        else:
            # General recovery
            return self.generate_general_recovery(failed_action, remaining_plan)

    def handle_navigation_failure(self, failed_action, remaining_plan):
        """
        Handle navigation failure
        """
        # Try alternative path
        alternative_paths = self.find_alternative_paths(failed_action)
        if alternative_paths:
            return alternative_paths[0] + remaining_plan
        else:
            return [("report_failure", {"reason": "No alternative path available"})]

    def handle_grasping_failure(self, failed_action, remaining_plan):
        """
        Handle grasping failure
        """
        # Try different grasp approach
        return [("adjust_grasp", {}), failed_action] + remaining_plan

    def handle_object_not_found(self, failed_action, remaining_plan):
        """
        Handle object not found failure
        """
        # Search for object in other locations
        search_actions = self.generate_search_plan(failed_action)
        return search_actions + [failed_action] + remaining_plan

Performance Optimization and Best Practices​

1. Plan Caching​

Cache frequently executed plans to improve efficiency:

class CachedTaskPlanner:
    def __init__(self, base_planner):
        self.base_planner = base_planner
        self.plan_cache = {}
        self.cache_size_limit = 100

    def plan_from_natural_language(self, command, world_state):
        """
        Plan with caching
        """
        cache_key = self.generate_cache_key(command, world_state)

        if cache_key in self.plan_cache:
            self.get_logger().info("Retrieved plan from cache")
            return self.plan_cache[cache_key]

        # Generate new plan
        plan = self.base_planner.plan_from_natural_language(command, world_state)

        # Add to cache
        if len(self.plan_cache) < self.cache_size_limit:
            self.plan_cache[cache_key] = plan

        return plan

    def generate_cache_key(self, command, world_state):
        """
        Generate cache key from command and simplified world state
        """
        # Simplify world state for caching purposes
        simplified_state = self.simplify_world_state(world_state)
        return f"{command}_{hash(str(simplified_state))}"

2. Hierarchical Plan Execution​

Execute plans in a hierarchical manner with proper error handling:

class HierarchicalExecutor:
    def __init__(self):
        self.executors = {
            "high_level": self.execute_high_level,
            "mid_level": self.execute_mid_level,
            "low_level": self.execute_low_level
        }

    def execute_plan(self, plan):
        """
        Execute plan hierarchically
        """
        for task in plan:
            level = self.determine_task_level(task)
            result = self.executors[level](task)
            if not result.success:
                return result
        return {"success": True, "message": "Plan completed"}

Future Directions and Research​

1. Learning from Execution​

Adapt planning based on execution experience:

class LearningPlanner:
    def __init__(self):
        self.execution_history = []
        self.success_rates = {}

    def update_from_execution(self, command, plan, result):
        """
        Update planner based on execution results
        """
        self.execution_history.append({
            "command": command,
            "plan": plan,
            "result": result,
            "timestamp": time.time()
        })

        # Update success rates
        plan_str = str(plan)
        if plan_str not in self.success_rates:
            self.success_rates[plan_str] = {"success": 0, "total": 0}

        self.success_rates[plan_str]["total"] += 1
        if result.get("success", False):
            self.success_rates[plan_str]["success"] += 1

class CollaborativePlanner:

        self.human_capabilities = {}

    def plan_collaborative_task(self, command, participants):

        Plan task involving multiple participants
        """
        # Assign subtasks based on capabilities
        task_allocation = self.allocate_tasks(command, participants)
